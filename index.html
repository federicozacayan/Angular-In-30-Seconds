    HttpClientModule
  ],

/* timer.service.ts */
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http'; //check it out
@Injectable({
  providedIn: 'root'
})
export class TimerService {
  constructor(private http: HttpClient) { }//check it out
  getCurrentDate(){
    //check it out
    return this.http.get('https://jsonplaceholder.typicode.com/todos/1');
  }
}

/* footer.component.ts */
import { Component, OnInit } from '@angular/core';
import { TimerService } from './../timer.service'
...
export class FooterComponent implements OnInit {
  date:string = '...'
  constructor(private timerService: TimerService) { }
  ngOnInit() {
    this.timerService.getCurrentDate().subscribe(data => {
      this.date = JSON.stringify(data);
      console.log(JSON.stringify(data))
    }, error => {
      console.error(error)
    });
  }
}

Two way binding

We can send prop from the component to the view, but to send data fron the view to the component we can use [(ngModel)].

<!-- home.component.html -->
<input [value]="name" (keyup.enter)="name = $event.target.value; onkeyUp()">
<input [(ngModel)]="lastName" (keyup.enter)="onkeyUp()">
/* home.component.ts */
...
export class HomeComponent implements OnInit {
  ...
  name:string = 'Jhon';
  lastName:string = 'Doe';
  onkeyUp(){
    console.log(this.name, this.lastName);
  }
